	\section{Question}
	\begin{frame}{Question 2(b)}
	If 
A =
\begin{bmatrix}
1 & 3\\
3 & 4\\
\end{bmatrix} and 

B =
\begin{bmatrix}
-2 & 1\\
-3 & 2\\
\end{bmatrix}

and $ A^2 - 5B^2 = 5C $ .Find matrix C where C is a 2 by 2 matrix.\\


	\end{frame}
	
	% The solution
	\section{Solution}
	\begin{frame}{Solution}
	A^2 &=\begin{bmatrix}
    1 & 3\\
    3 & 4\\
    \end{bmatrix}\times
    \begin{bmatrix}
    1 & 3\\
    3 & 4\\
    \end{bmatrix}\\
    
A^2 &=\begin{bmatrix}
    10 & 15\\
    15 & 25\\
    \end{bmatrix}\\    
    
B^2 &=\begin{bmatrix}
    -2 & 1\\
    -3 & 2\\
    \end{bmatrix}\times
    \begin{bmatrix}
    -2 & 1\\
    -3 & 2\\
    \end{bmatrix}\\
    
B^2 &=\begin{bmatrix}
    1 & 0\\
    0 & 1\\
    \end{bmatrix}\\     

	\end{frame}
		
	\begin{frame}{}
	
5B^2 &=\begin{bmatrix}
    5 & 0\\
    0 & 5\\
    \end{bmatrix}\\     
    
5C &= A^2 - 5B^2\\

  5C &=\begin{bmatrix}
    10 & 15\\
    15 & 25\\
    \end{bmatrix} - \begin{bmatrix}
    5 & 0\\
    0 & 5\\
    \end{bmatrix}\\
    
5C &=\begin{bmatrix}
    5 & 15\\
    15 & 20\\
    \end{bmatrix}\\

 
	\end{frame}
	
	% The final answer
	\section{Answer}
	\begin{frame}{Answer}
	\begin{align}
C  &=\begin{bmatrix}
    1 & 3\\
    3 & 4\\
    \end{bmatrix}\\

    \end{align}     
	\end{frame}
